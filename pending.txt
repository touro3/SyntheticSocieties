    tmux new-session -d -s bgf_ablation \
      "cd /mnt/sdb1/workspace/lucastourinho/SyntheticSocieties && \
       source venv/bin/activate && \
       for MODE in no_persona minimal_persona rich_persona no_memory no_network no_institutions; do \
         for SEED in 42 123 7; do \
           python scripts/run_config_simulation.py \
             --config configs/base_config.yaml \
             policy.type=ablated_llm \
             ablation.mode=\$MODE \
             project.experiment_id=ablation_\${MODE}_s\${SEED} \
             project.seed=\$SEED \
             simulation.rounds=10 \
             simulation.population_size=20; \
         done; \
       done 2>&1 | tee logs/ablation_run.log; echo '=== DONE ==='"

    # Session 2: abl_v* ablation ladder (V0–V5 × 3 seeds × 20 agents × 10 rounds)
    tmux new-session -d -s bgf_ladder \
      "cd /mnt/sdb1/workspace/lucastourinho/SyntheticSocieties && \
       source venv/bin/activate && \
       python scripts/run_full_pipeline.py \
         --include-llm --run-ablation-ladder \
         --seeds 42,123,7 --rounds 10 --agents 20 \
         2>&1 | tee logs/ladder_run.log; echo '=== DONE ==='"

    WHY FIRST: Needed to regenerate ladder_ablation.png, ablation_effect.png,
               and grafo_A_ablated.png — explicitly flagged as "plots that will change."
    LOAD:      6 modes × 3 seeds × 200 agent-decisions = 3,600 decisions (Session 1)
               6 levels × 3 seeds × 200 decisions = 3,600 decisions (Session 2)
    ESTIMATE:  ~25–40 min each (run in parallel → wall-clock ~35 min)
